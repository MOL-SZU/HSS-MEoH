```python
import numpy as np
import random
import pygmo as pg

def HSS(points, k: int, reference_point=None, T_init: float = 1.0, T_min: float = 1e-6, 
        alpha: float = 0.95, max_iter: int = 1000, swap_prob: float = 0.7) -> np.ndarray:
    n, m = points.shape
    
    if reference_point is None:
        reference_point = np.max(points, axis=0) * 1.1
    
    # 初始化：随机选择 k 个点
    selected_indices = set(np.random.choice(n, size=k, replace=False))
    
    # 计算当前超体积
    def compute_hv(indices):
        if not indices:
            return 0.0
        selected_points = points[list(indices)]
        hv = pg.hypervolume(selected_points)
        return hv.compute(reference_point)
    
    current_hv = compute_hv(selected_indices)
    
    # 模拟退火参数（使用传入的参数）
    T = T_init
    iteration = 0
    
    while T > T_min and iteration < max_iter:
        # 随机选择操作类型：交换或替换（使用swap_prob参数）
        if random.random() < swap_prob:  # swap_prob概率进行交换
            # 随机交换一个已选点和一个未选点
            if len(selected_indices) < n:
                selected_list = list(selected_indices)
                unselected = [i for i in range(n) if i not in selected_indices]
                
                if selected_list and unselected:
                    s = random.choice(selected_list)
                    r = random.choice(unselected)
                    
                    new_indices = selected_indices.copy()
                    new_indices.remove(s)
                    new_indices.add(r)
                    
                    new_hv = compute_hv(new_indices)
                    
                    # 接受准则
                    delta = new_hv - current_hv
                    if delta > 0 or random.random() < np.exp(delta / T):
                        selected_indices = new_indices
                        current_hv = new_hv
        
        else:  # (1-swap_prob)概率进行替换
            # 随机移除一个点，然后贪心添加最佳点
            if selected_indices:
                # 随机移除一个点
                s = random.choice(list(selected_indices))
                temp_indices = selected_indices.copy()
                temp_indices.remove(s)
                
                # 贪心添加最佳未选点
                unselected = [i for i in range(n) if i not in temp_indices]
                best_idx = -1
                best_hv = -np.inf
                
                for r in unselected:
                    test_indices = temp_indices.copy()
                    test_indices.add(r)
                    test_hv = compute_hv(test_indices)
                    
                    if test_hv > best_hv:
                        best_hv = test_hv
                        best_idx = r
                
                if best_idx != -1:
                    temp_indices.add(best_idx)
                    new_hv = best_hv
                    
                    # 接受准则
                    delta = new_hv - current_hv
                    if delta > 0 or random.random() < np.exp(delta / T):
                        selected_indices = temp_indices
                        current_hv = new_hv
        
        # 降温
        T *= alpha
        iteration += 1
    
    # 最终局部搜索：尝试所有单点交换
    improved = True
    while improved:
        improved = False
        selected_list = list(selected_indices)
        unselected = [i for i in range(n) if i not in selected_indices]
        
        for s in selected_list:
            for r in unselected:
                new_indices = selected_indices.copy()
                new_indices.remove(s)
                new_indices.add(r)
                
                new_hv = compute_hv(new_indices)
                
                if new_hv > current_hv:
                    selected_indices = new_indices
                    current_hv = new_hv
                    improved = True
                    break
            
            if improved:
                break
    
    # 返回选中的点
    return points[list(selected_indices)]
```

```python
parameter_ranges = {
    'T_init': (0.1, 10.0),      # 初始温度：控制搜索的初始随机性
    'T_min': (1e-8, 1e-3),      # 最小温度：决定算法何时停止退火
    'alpha': (0.85, 0.999),     # 退火率：控制温度下降速度
    'max_iter': (100, 5000),    # 最大迭代次数：控制计算时间
    'swap_prob': (0.3, 0.9)     # 交换操作概率：平衡探索与利用
}
```
