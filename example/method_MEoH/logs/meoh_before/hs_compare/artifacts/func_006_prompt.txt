        return hv.compute(reference_point)
    
    def hypervolume_contribution(point, subset, ref):
        if len(subset) == 0:
            return np.prod(ref - point)

        hv_with = HV_cal(combined, ref)
        hv_without = HV_cal(subset, ref)
        return hv_with - hv_without

        A[:, 0] = np.min(dist, axis=1)
        A[:, 1] = np.argmin(dist, axis=1)
        return A
    
    def association_upd(data, dataSel, i, newS, A):
